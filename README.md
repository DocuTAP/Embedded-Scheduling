<!DOCTYPE html>

  <head>
    <script>
      var _clockwise = {
        host: 'www.clockwisemd.com',
        group_id: <GROUP_ID>,
        token: <API_ACCESS_TOKE>,
        filters: ["reasons", "location", "physician"],
        showHeader: true
      };
    </script>
    <script src="//www.clockwisemd.com/groups/<GROUP_ID>/schedule.js"></script>
  </head>
  <body></body>
